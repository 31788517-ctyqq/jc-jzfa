### 4.3 fetch-match-list 云函数

```javascript
/**
 * 获取指定日期的比赛列表
 * 
 * 入参: { date: "2026-05-18" (可选，默认当天) }
 * 出参: { code: 1, data: [match objects] }
 */
exports.main = async (event, context) => {
  const axios = require('axios');
  const { date } = event;
  const targetDate = date || new Date().toISOString().slice(0, 10);
  
  // 1. 先获取token
  const loginRes = await callFunction('login-midou');
  if (loginRes.code !== 1) return loginRes;
  const token = loginRes.data.token;
  
  // 2. 获取比赛列表
  const timestamp = new Date(targetDate).getTime();
  const listUrl = 'https://midou310.com/mdsj/score/footballDataList.do';
  const resp = await axios.get(listUrl, {
    params: {
      time: timestamp,
      order: ' status desc, start_datetime asc, data_id asc '
    },
    headers: { Cookie: `token=${token}` },
    timeout: 15000
  });
  
  if (resp.data.code !== 1) {
    return { code: 0, msg: '获取比赛列表失败' };
  }
  
  // 3. 格式化数据
  const matches = resp.data.data.map(m => ({
    matchId: m.matchId,
    num: m.num,
    homeName: m.homeName,
    visitName: m.visitName,
    leagueName: m.leagueName,
    startTime: m.startTime,
    startDatetime: `${targetDate} ${m.startTime}`,
    matchStatus: m.matchStatus,
    score: m.score || '',
    halfScore: m.halfScore || '',
    duration: m.duration || '',
    recommNum: m.recommNum || 0,
    bigDataNum: m.bigDataNum || 0,
    liveNum: m.liveNum || 0,
    date: targetDate,
    captureTime: new Date().toISOString()
  }));
  
  return { code: 1, data: matches, today: resp.data.today, hour: resp.data.hour };
};
```

### 4.4 fetch-recommend 云函数

```javascript
/**
 * 获取单场比赛的推荐详情
 * 
 * 入参: { matchId: "2039816" }
 * 出参: {
 *   code: 1,
 *   data: {
 *     matchId, homeName, visitName,
 *     captureTime,
 *     matchStatus,   // 抓取时的比赛状态
 *     recommendations: [{ type, num, result }],
 *     rawData: {...}
 *   }
 * }
 */
exports.main = async (event, context) => {
  const axios = require('axios');
  const { matchId } = event;
  if (!matchId) return { code: 0, msg: '缺少matchId' };
  
  // 1. 获取token
  const loginRes = await callFunction('login-midou');
  if (loginRes.code !== 1) return loginRes;
  const token = loginRes.data.token;
  
  // 2. 获取推荐数据
  const recommUrl = 'https://midou310.com/mdsj/score/getExpertRecommData.do';
  const resp = await axios.get(recommUrl, {
    params: { dataId: matchId, type: 0 },
    headers: { Cookie: `token=${token}` },
    timeout: 10000
  });
  
  if (resp.data.code !== 1) {
    return { code: 0, msg: `获取推荐失败: ${resp.data.msg}` };
  }
  
  // 3. 提取推荐方向汇总（过滤掉底部的mapData和newRecomm）
  const rawData = resp.data.data;
  const recommendations = rawData
    .filter(item => item.type)  // 过滤掉最后一条无type的数据
    .map(item => ({
      type: item.type,
      num: item.num,
      result: item.result !== undefined ? item.result : null
    }));
  
  return {
    code: 1,
    data: {
      matchId,
      captureTime: new Date().toISOString(),
      captureTimestamp: Date.now(),
      recommendations,
      rawData
    }
  };
};
```

 */
exports.main = async (event, context) => {

  const db = cloud.database();
  const _ = db.command;
  const dateStr = new Date().toISOString().slice(0, 10);
  
  try {
    // Step 1: 获取比赛列表
    const matchListRes = await callFunction('fetch-match-list', { date: dateStr });
    if (matchListRes.code !== 1) throw new Error(matchListRes.msg);
    const matches = matchListRes.data;
    
    // Step 2: 更新 matches 集合（覆盖写入当天数据）
    const batch = db.collection('matches');
    // 先清除当天旧数据，再批量插入
    await batch.where({ date: dateStr }).remove();
    
    const matchIds = [];
    for (const match of matches) {
      await batch.add({
        data: { _id: match.matchId, ...match }
      });
      matchIds.push(match.matchId);
    }
    
    // Step 3: 遍历每场比赛获取推荐详情
    let successCount = 0;
    let failCount = 0;
    const errors = [];
    
    for (const matchId of matchIds) {
      try {
        const recommRes = await callFunction('fetch-recommend', { matchId });
        if (recommRes.code === 1) {
          // 写入 recommmends 集合
          const doc = {
            _id: `${matchId}_${dateStr}_${new Date().toISOString().slice(11, 16).replace(':', '')}`,
            matchId,
            date: dateStr,
            captureTime: new Date().toISOString(),
            captureTimestamp: Date.now(),
            recommendations: recommRes.data.recommendations,
            rawData: recommRes.data.rawData
          };
          await db.collection('recommends').add({ data: doc });
          successCount++;
        } else {
          failCount++;
          errors.push({ matchId, error: recommRes.msg });
        }
      } catch (err) {
        failCount++;
        errors.push({ matchId, error: err.message });
      }
    }
    
    // Step 4: 记录抓取日志
    const duration = Date.now() - startTime;
    await db.collection('crawl_logs').add({
      data: {
        captureTime: new Date().toISOString(),
        captureTimestamp: Date.now(),
        matchCount: matches.length,
        successCount,
        failCount,
        errors,
        duration
      }
    });
    
    return {
      code: 1,
      data: { matchCount: matches.length, successCount, failCount, duration }
    };
  } catch (err) {
    return { code: 0, msg: `定时抓取失败: ${err.message}` };
  }
};

### 4.6 calc-daily-hit-rate 云函数

```javascript
/**
 * 每日凌晨计算命中率
 * 
 * Cron: 0 0 2 * * * * (每天凌晨2点)
 * 
 * 算法：
 * 1. 遍历最近30天
 * 2. 获取每天已完成比赛(matchStatus >= 2)
 * 3. 获取每场比赛最后一次抓取的推荐数据
 * 4. 按推荐方向汇总命中/未中
 * 5. 写入 hit_rates 集合
 */
exports.main = async (event, context) => {
  const db = cloud.database();
  const _ = db.command;
  
  const today = new Date();
  const results = [];
  
  // 遍历30天
  for (let i = 0; i < 30; i++) {
    const day = new Date(today);
    day.setDate(day.getDate() - i);
    const dateStr = day.toISOString().slice(0, 10);
    
    // 获取该天所有已完成比赛（从matches集合）
    const matchRes = await db.collection('matches')
      .where({
        date: dateStr,
        matchStatus: _.gte(2)  // 已完成
      })
      .get();
    
    const completedMatches = matchRes.data || [];
    if (completedMatches.length === 0) continue;
    
    // 方向汇总 Map<direction, { total, hit, miss }>
    const directionMap = new Map();
    
    for (const match of completedMatches) {
      // 获取该比赛最后一次抓取的推荐数据
      const recommRes = await db.collection('recommends')
        .where({ matchId: match._id })
        .orderBy('captureTimestamp', 'desc')
        .limit(1)
        .get();
      
      if (recommRes.data.length === 0) continue;
      const lastRecomm = recommRes.data[0];
      
      for (const r of lastRecomm.recommendations) {
        if (!directionMap.has(r.type)) {
          directionMap.set(r.type, { total: 0, hit: 0, miss: 0 });
        }
        const stat = directionMap.get(r.type);
        // 只统计有专家推荐的(num > 0)且有结果(result != null)的
        if (r.num > 0 && r.result !== null && r.result !== undefined) {
          stat.total += r.num;
          if (r.result === 1) {
            stat.hit += r.num;
          } else if (r.result === 0) {
            stat.miss += r.num;
          }
        }
      }
    }
    
    // 保存该天的命中率
    if (directionMap.size > 0) {
      const directionStats = [];
      for (const [direction, stats] of directionMap) {
        directionStats.push({
          direction,
          totalRecommends: stats.total,
          hitCount: stats.hit,
          missCount: stats.miss,
          hitRate: stats.total > 0 ? Number((stats.hit / stats.total * 100).toFixed(2)) : 0
        });
      }
      
      results.push({
        _id: dateStr,
        statDate: dateStr,
        totalMatches: completedMatches.length,
        completedMatches: completedMatches.length,
        lastCalcTime: new Date().toISOString(),
        directionStats
      });
    }
  }
  
  // 批量写入/更新 hit_rates 集合
  for (const doc of results) {
    const existing = await db.collection('hit_rates').doc(doc._id).get();
    if (existing.data) {
      await db.collection('hit_rates').doc(doc._id).update({ data: doc });
    } else {
      await db.collection('hit_rates').add({ data: doc });
    }
  }
  
  return { code: 1, data: { daysCalculated: results.length } };
};
```

### 4.7 HTTP API 云函数

```javascript
/**
 * 小程序前端调用的统一入口
 * 
 * HTTP触发路径: /api/{action}
 * 
 * action 列表:
 *   - match-list:     获取当日/指定日期比赛列表
 *   - recommend-trend: 获取比赛推荐趋势数据
 *   - hit-rate-stats:  获取命中率统计数据
 *   - match-detail:    获取单场比赛基本信息
 *   - ranking-list:    获取当天推荐排名（含方向筛选）
 */

exports.main = async (event, context) => {
  const db = cloud.database();
  const _ = db.command;
  const { action, data } = event;
  
  switch (action) {
    case 'match-list': {
      // 入参: { date: "2026-05-18" }
      const date = data.date || new Date().toISOString().slice(0, 10);
      const res = await db.collection('matches')
        .where({ date })
        .orderBy('num', 'asc')
        .get();
      return { code: 1, data: res.data };
    }
    
    case 'recommend-trend': {
      // 入参: { matchId: "2039816" }
      const { matchId } = data;
      const res = await db.collection('recommends')
        .where({ matchId })
        .orderBy('captureTimestamp', 'asc')
        .get();
      
      // 整理成图表需要的格式
      const trendData = res.data.map(doc => ({
        captureTime: doc.captureTime,
        captureTimestamp: doc.captureTimestamp,
        recommendations: doc.recommendations,
        matchStatus: doc.matchStatus
      }));
      
      // 提取所有推荐方向
      const allDirections = new Set();
      trendData.forEach(t => {
        t.recommendations.forEach(r => {
          if (r.num > 0) allDirections.add(r.type);
        });
      });
      
      // 组装图表序列
      const series = [];
      const directions = Array.from(allDirections);
      for (const dir of directions) {
        series.push({
          name: dir,
          type: 'line',
          smooth: true,
          data: trendData.map(t => {
            const found = t.recommendations.find(r => r.type === dir);
            return found ? found.num : 0;
          })
        });
      }
      
      return {
        code: 1,
        data: {
          matchId,
          timeLabels: trendData.map(t => t.captureTime.slice(11, 16)),
          series,
          // 最后一次抓取的命中结果
          lastResult: trendData.length > 0 ? trendData[trendData.length - 1].recommendations : []
        }
      };
    }
    
    case 'hit-rate-stats': {
      // 入参: { days: 30 }
      const days = data.days || 30;
      const endDate = new Date();
      const startDate = new Date();
      startDate.setDate(startDate.getDate() - days);
      const startStr = startDate.toISOString().slice(0, 10);
      const endStr = endDate.toISOString().slice(0, 10);
      
      // 获取最近N天的命中率数据
      const res = await db.collection('hit_rates')
        .where({
          _id: _.gte(startStr).and(_.lte(endStr))
        })
        .orderBy('statDate', 'desc')
        .limit(days)
        .get();
      
      // 汇总所有方向数据
      const directionAgg = new Map();
      const dailyTrend = [];
      
      for (const doc of res.data) {
        dailyTrend.push({
          date: doc.statDate,
          totalMatches: doc.completedMatches,
          directions: doc.directionStats
        });
        
        for (const ds of doc.directionStats) {
          if (!directionAgg.has(ds.direction)) {
            directionAgg.set(ds.direction, { total: 0, hit: 0, miss: 0 });
          }
          const stat = directionAgg.get(ds.direction);
          stat.total += ds.totalRecommends;
          stat.hit += ds.hitCount;
          stat.miss += ds.missCount;
        }
      }
      
      // 计算总命中率
      const directionStats = [];
      for (const [direction, stats] of directionAgg) {
        directionStats.push({
          direction,
          totalRecommends: stats.total,
          hitCount: stats.hit,
          missCount: stats.miss,
          hitRate: stats.total > 0 ? Number((stats.hit / stats.total * 100).toFixed(2)) : 0
        });
      }
      
      // 按命中率降序
      directionStats.sort((a, b) => b.hitRate - a.hitRate);
      
      return {
        code: 1,
        data: {
          totalDays: dailyTrend.length,
          directionStats,
          dailyTrend: dailyTrend.reverse()  // 按日期升序
        }
      };
    }

    case 'ranking-list': {
      // 入参: { date: "2026-05-18", direction: "胜" }
      const rankingDate = data.date || new Date().toISOString().slice(0, 10);
      const filterDirection = data.direction || null;
      const matchRes = await db.collection('matches').where({ date: rankingDate }).get();
      const matches = matchRes.data || [];
      const rankingList = [];
      for (const match of matches) {
        const r = await db.collection('recommends')
          .where({ matchId: match._id }).orderBy('captureTimestamp', 'desc').limit(1).get();
        if (r.data.length === 0) continue;
        const latest = r.data[0];
        if (filterDirection) {
          const found = latest.recommendations.find(x => x.type === filterDirection);
          if (found && found.num > 0) rankingList.push({ matchId: match._id, homeName: match.homeName, visitName: match.visitName, direction: filterDirection, expertCount: found.num });
        } else {
          const maxDir = latest.recommendations.reduce((a, b) => (b.num || 0) > ((a && a.num) || 0) ? b : a, null);
          if (maxDir && maxDir.num > 0) rankingList.push({ matchId: match._id, homeName: match.homeName, visitName: match.visitName, direction: maxDir.type, expertCount: maxDir.num });
        }
      }
      rankingList.sort((a, b) => b.expertCount - a.expertCount);
      const result = rankingList.map((item, i) => ({ rank: i + 1, ...item }));
      // 提取可用方向
      const allDirs = new Set();
      for (const m of matches) {
        const r2 = await db.collection('recommends').where({ matchId: m._id }).orderBy('captureTimestamp', 'desc').limit(1).get();
        if (r2.data.length > 0) r2.data[0].recommendations.forEach(x => { if (x.num > 0) allDirs.add(x.type); });
      }
      return { code: 1, data: { date: rankingDate, filterDirection, totalMatches: result.length, availableDirections: Array.from(allDirs).sort(), ranking: result } };
    }

    default:
      return { code: 0, msg: `未知 action: ${action}` };
  }
};
```

### 5.4 视觉与交互设计规范（避免 AI 模板化）

> 本规范用于指导页面视觉设计，确保 UI 具有**人工设计的品质感**，避免出现 AI 生成的"模板化"外观（典型的 Bootstrap 感、生硬布局、缺乏细节等）。

#### 5.4.1 设计原则

| 原则 | 说明 | 反例（AI 典型问题） |
|------|------|---------------------|
| **呼吸感** | 元素之间保留充足间距，不堆砌信息 | 卡片紧贴、文字挤在一起、边距不一致 |
| **层次感** | 通过字号、字重、颜色区分信息层级 | 所有文字一个大小、没有视觉权重 |
| **一致性** | 同类型组件在全局保持统一的外观和行为 | 同是按钮，有的圆角 4px 有的 8px |
| **细节感** | 添加微妙的阴影、圆角、过渡动画 | 卡片纯白硬边、点击无反馈、切换生硬 |
| **品牌感** | 统一的品牌色系统，贯穿全局 | 颜色随意使用、五颜六色没有主色调 |
| **上下文** | 图表和列表周围有说明文本和标题 | 光秃秃一个图表，没有标题说明 |

## 8. 核心算法设计

### 8.1 推荐趋势图表数据组装

```
输入: recommends 集合中 matchId=xx 的所有文档（按时间升序）

算法:
1. 遍历所有文档，提取所有出现过的推荐方向
2. 对于每个时间点，记录每个方向的专家数量
3. 整理为 ECharts 需要的格式

输出格式:
{
  timeLabels: ["11:00", "11:30", "12:00", ...],
  series: [
    { name: "胜", type: "line", data: [5, 6, 6, ...] },
    { name: "平", type: "line", data: [3, 3, 2, ...] },
    { name: "负", type: "line", data: [3, 4, 4, ...] }
  ],
  lastResult: [
    { type: "胜", num: 6, result: 1 },
    { type: "平", num: 3, result: 0 }
  ]
}
```

### 8.2 命中率统计算法

```
输入: 近30天的 recommends 集合中 matchStatus>=2 的文档

算法:
1. 按天遍历30天
2. 每天中，对每场已完成比赛，取最后一次抓取的推荐数据
3. 对每个推荐方向：
   - total = 各场比赛该方向专家人数之和
   - hit = 各场比赛中 result=1 的专家人数之和
   - miss = 各场比赛中 result=0 的专家人数之和
   - hitRate = hit / total * 100

输出:
{
  directionStats: [
    { direction: "胜", totalRecommends: 120, hitCount: 72, missCount: 48, hitRate: 60.00 },
    ...
  ],
  dailyTrend: [
    { date: "2026-05-18", directions: [ 每日各方向命中率 ] },
    ...
  ]
}
```

### 8.3 命中标记判定

```
输入: 推荐API返回的 result 字段
         result=1  → 命中  → 绿色标记 "✓ 中"
         result=0  → 未中  → 灰色标记 "✗ 未中"
         result=null/undefined → 待开 → 灰色标记 "— 待开"

比赛级别的命中汇总:
   - 遍历该场比赛所有推荐方向
   - 如果任何方向 result=1 → 该比赛部分推荐命中
   - 如果所有方向 result=0 → 该比赛推荐未中
   - 如果所有方向 result=null → 待开
```

### 8.4 推荐排名算法

```
输入: 当天比赛列表 + 最新一次 recommends 数据

综合排名算法:
1. 获取当天 matches 集合中所有比赛
2. 对每场比赛，获取 recommends 集合中最新一次抓取数据
3. 从该数据中找出 expertCount 最大的推荐方向
4. 记录: { matchId, homeName, visitName, direction(最大方向), expertCount, matchStatus }
5. 按 expertCount 降序排列所有比赛
6. 生成排名序号 (1, 2, 3...)

按方向筛选排名算法:
1. 获取当天 matches 集合中所有比赛
2. 对每场比赛，获取 recommends 集合中最新一次抓取数据
3. 从中找出 direction = filterDirection 的数据
4. 如果存在且 expertCount > 0，则记录
5. 按 expertCount 降序排列
6. 生成排名序号

可用方向提取:
1. 遍历当天所有比赛的最新 recommends 数据
2. 收集所有 num > 0 的方向
3. 去重后排序，作为方向筛选器的选项

输出格式:
{
  date: "2026-05-18",
  filterDirection: null (或 "胜"),
  totalMatches: 8,
  availableDirections: ["胜","平","负","胜平","让胜","让负","总进球-2、3球",...],
  ranking: [
    { rank: 1, matchId: "2039816", homeName: "拉赫蒂", visitName: "瓦萨",
      direction: "胜", expertCount: 6 },
    ...
  ]
}
```

### 9.3 错误码定义

| 错误码 | 含义 | 说明 |
|--------|------|------|
| `1` | 成功 | 请求成功，`data` 字段包含返回数据 |
| `0` | 通用错误 | 业务处理失败，`msg` 提供错误描述 |
| `101` | 参数缺失 | 必填参数未传入（如缺少 matchId） |
| `102` | 登录失败 | 米斗数据登录失败（token过期/密码错误） |
| `103` | 数据不存在 | 请求的数据（比赛/推荐）不存在 |
| `104` | 外部API超时 | 米斗数据接口请求超时 |
| `105` | 数据库错误 | 云数据库读写异常 |
| `106` | 频率限制 | 请求过于频繁，请稍后重试 |
| `500` | 服务器内部错误 | 未预期的异常，查看云函数日志 |

### 10.2 Cron 策略说明

| Cron表达式 | 含义 |
|-----------|------|
| `0 */30 11-23 * * * *` | 每天 11:00~23:59，每隔 30 分钟触发一次（11:00、11:30、12:00...23:30） |
| `0 0 2 * * * *` | 每天凌晨 2:00 触发一次 |

### 10.3 抓取结束策略

- 某场比赛 `matchStatus >= 2`（已完成/取消/延期）后，该场比赛不再抓取
- 每天所有比赛都完成时，停止当天抓取
- 23:30 最后一次抓取，此后即使有比赛未完成也不继续（比赛一般23点前结束）

#### 14.2.1 语法规范

| 规则 | 要求 | 正确示例 | 错误示例 |
|------|------|---------|---------|
| 变量声明 | 统一 `const` / `let`，禁止 `var` | `const matchId = '123'` | `var matchId = '123'` |

| 默认参数 | 使用函数默认值 | `function(a = 1)` | `a = a || 1` |
| 展开运算符 | 数组/对象拷贝使用 | `[...arr]`, `{...obj}` | `Object.assign({}, obj)` |
